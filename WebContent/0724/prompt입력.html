실행시 이름을 prompt()를 이용하여 입력하고
	결과를 div에 출력한다.
	
	window.onload = function(){
		var name = prompt("이름을 입력하세요.", "한글 또는 영문");
		
		var vdiv = document.getElementById('out');
		vdiv.innerHTML = "이름 : " + name;	
	}

확인버튼을 클릭하면 두개의 수를 입력하여 덧셈을 하고 div에 출력

	function add(){
		var x = prompt("첫번째 정수를 입력");
		var y = prompt("두번째 정수를 입력");
		
		var result = parseInt(x) + parseInt(y);
		
		var vdiv2 = document.getElementById('sum');
		vdiv2.innerHTML = x + " + " + y + " = " + result;
	}
